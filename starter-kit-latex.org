<div class="highlight highlight-source-lisp"><pre>  <span class="pl-c">;; Biber under AUCTeX</span>
  (<span class="pl-k">defun</span> <span class="pl-en">TeX-run-Biber</span> (name command file)
    <span class="pl-s"><span class="pl-pds">"</span>Create a process for NAME using COMMAND to format FILE with Biber.<span class="pl-pds">"</span></span> 
   (<span class="pl-k">let</span> ((process (TeX-run-command name command file)))
      (<span class="pl-c1">setq</span> TeX-sentinel-function 'TeX-Biber-sentinel)
      (<span class="pl-k">if</span> TeX-process-asynchronous
          process
        (TeX-synchronous-sentinel name file process))))
  
  (<span class="pl-k">defun</span> <span class="pl-en">TeX-Biber-sentinel</span> (process name)
    <span class="pl-s"><span class="pl-pds">"</span>Cleanup TeX output buffer after running Biber.<span class="pl-pds">"</span></span>
    (goto-char (point-max))
    (<span class="pl-c1">cond</span>
     <span class="pl-c">;; Check whether Biber reports any warnings or errors.</span>
     ((re-search-backward (concat
                           <span class="pl-s"><span class="pl-pds">"</span>^(There <span class="pl-cce">\\</span>(?:was<span class="pl-cce">\\</span>|were<span class="pl-cce">\\</span>) <span class="pl-cce">\\</span>([0-9]+<span class="pl-cce">\\</span>) <span class="pl-pds">"</span></span>
                           <span class="pl-s"><span class="pl-pds">"</span><span class="pl-cce">\\</span>(warnings?<span class="pl-cce">\\</span>|error messages?<span class="pl-cce">\\</span>))<span class="pl-pds">"</span></span>) <span class="pl-c1">nil</span> <span class="pl-c1">t</span>)
      <span class="pl-c">;; Tell the user their number so that she sees whether the</span>
      <span class="pl-c">;; situation is getting better or worse.</span>
      (message (concat <span class="pl-s"><span class="pl-pds">"</span>Biber finished with %s %s. <span class="pl-pds">"</span></span>
                       <span class="pl-s"><span class="pl-pds">"</span>Type `%s' to display output.<span class="pl-pds">"</span></span>)
               (match-string <span class="pl-c1">1</span>) (match-string <span class="pl-c1">2</span>)
               (substitute-command-keys
                <span class="pl-s"><span class="pl-pds">"</span><span class="pl-cce">\\\\</span>[TeX-recenter-output-buffer]<span class="pl-pds">"</span></span>)))
